# CORTHEX API 호출 예시
import requests

headers = {
  "Authorization": "Bearer sk-cx-****...****3a2f",
  "Content-Type": "application/json"
}

payload = {
  "command": "오늘 시장 동향 분석해줘",
  "agent_id": "agent_abc123"
}

response = requests.post(
  "https://api.corthex.io/v1/commands",
  headers=headers,
  json=payload
)

print(response.json())
